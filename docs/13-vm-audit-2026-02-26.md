import os, time

app = FastAPI()
TOKEN = os.environ.get("WORKER_TOKEN","")

@app.get("/health")
def health():
    return {"ok": True, "ts": int(time.time())}

@app.post("/run")

    if not TOKEN:
        raise HTTPException(500, "WORKER_TOKEN not set")
